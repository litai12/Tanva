# 租户基座（Tenancy Foundation）Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 把单租户后端改造成企业多租户基座——按域名解析租户、查询层自动隔离、认证/异步/实时通道绑定租户、存量数据迁移到主站。

**Architecture:** `Tenant.id` 为唯一真值，`TenantDomain` 域名映射。请求经 Host 中间件解析 tenantId 存入 AsyncLocalStorage(nestjs-cls)；Prisma `$extends` 查询扩展从 CLS 读 tenantId 自动注入 `where`/`data`；异步入口（支付回调/cron/worker）用 `runAsTenant(row.tenantId)` 显式切上下文，禁止默认主站。定价/套餐表进全局白名单基座阶段全平台共享。

**Tech Stack:** NestJS(Fastify adapter) + Prisma + PostgreSQL + nestjs-cls + passport-jwt。

**Spec:** `docs/superpowers/specs/2026-06-08-租户基座-design.md`

**约定：** 每个 Task 末尾 commit。所有命令在 `backend/` 下执行。后端测试框架为 Jest（`npm test`）。类型检查 `npm run build`。

```typescript
// backend/src/tenancy/tenant-context.service.ts
import { Injectable } from '@nestjs/common';
import { ClsService } from 'nestjs-cls';
import {
  CLS_PLATFORM_MODE_KEY,
  CLS_TENANT_KEY,
  PLATFORM_TENANT_ID,
} from './tenant.constants';

@Injectable()
export class TenantContextService {
  constructor(private readonly cls: ClsService) {}

  /** 当前请求租户 id；未设置时回落主站（仅用于读路径；写/异步入口必须显式 set） */
  getTenantId(): string {
    return this.cls.get(CLS_TENANT_KEY) ?? PLATFORM_TENANT_ID;
  }

  setTenantId(tenantId: string): void {
    this.cls.set(CLS_TENANT_KEY, tenantId);
  }

  isPlatformMode(): boolean {
    return this.cls.get(CLS_PLATFORM_MODE_KEY) === true;
  }

  /** 在指定租户上下文内执行（异步入口/超管精确操作单租户用） */
  async runAsTenant<T>(tenantId: string, fn: () => Promise<T> | T): Promise<T> {
    return this.cls.run(async () => {
      this.cls.set(CLS_TENANT_KEY, tenantId);
      this.cls.set(CLS_PLATFORM_MODE_KEY, false);
      return fn();
    });
  }

  /** 逃生舱：平台态执行，Prisma 扩展不注入 tenantId（仅 platform-admin / 异步定位订单用） */
  async runAsPlatform<T>(fn: () => Promise<T> | T): Promise<T> {
    return this.cls.run(async () => {
      this.cls.set(CLS_PLATFORM_MODE_KEY, true);
      return fn();
    });
  }
}
```

```sql
DROP INDEX IF EXISTS "User_phone_key";
DROP INDEX IF EXISTS "User_email_key";
DROP INDEX IF EXISTS "User_watchaUserId_key";
DROP INDEX IF EXISTS "User_wechatOfficialOpenId_key";
DROP INDEX IF EXISTS "User_wechatUnionId_key";
CREATE UNIQUE INDEX IF NOT EXISTS "User_tenantId_phone_key" ON "User"("tenantId","phone");
CREATE UNIQUE INDEX IF NOT EXISTS "User_tenantId_email_key" ON "User"("tenantId","email");
CREATE UNIQUE INDEX IF NOT EXISTS "User_tenantId_watchaUserId_key" ON "User"("tenantId","watchaUserId");
CREATE UNIQUE INDEX IF NOT EXISTS "User_tenantId_wechatOfficialOpenId_key" ON "User"("tenantId","wechatOfficialOpenId");
CREATE UNIQUE INDEX IF NOT EXISTS "User_tenantId_wechatUnionId_key" ON "User"("tenantId","wechatUnionId");
```

```typescript
// backend/src/prisma/tenant-prisma.extension.ts
import { Prisma } from '@prisma/client';
import { TENANT_GLOBAL_MODELS } from '../tenancy/tenant.constants';

export interface TenantCtx {
  tenantId: string;
  isPlatform: boolean;
}

const READ_OPS = new Set([
  'findMany', 'findFirst', 'findFirstOrThrow', 'count', 'aggregate', 'groupBy',
  'updateMany', 'deleteMany',
]);

/** 纯函数：根据 model/operation/ctx 决定如何注入 tenantId，便于单测 */
export function buildTenantArgs(model: string, operation: string, args: any, ctx: TenantCtx): any {
  const next = { ...(args ?? {}) };
  if (ctx.isPlatform) return next;            // 逃生舱：不注入
  if (TENANT_GLOBAL_MODELS.has(model)) return next; // 白名单：不注入

  const t = ctx.tenantId;
  if (operation === 'create') {
    next.data = { ...(next.data ?? {}), tenantId: t };
  } else if (operation === 'createMany') {
    const data = next.data;
    next.data = Array.isArray(data)
      ? data.map((d: any) => ({ ...d, tenantId: t }))
      : { ...(data ?? {}), tenantId: t };
  } else if (operation === 'upsert') {
    next.where = { ...(next.where ?? {}), tenantId: t };
    next.create = { ...(next.create ?? {}), tenantId: t };
    next.update = { ...(next.update ?? {}), tenantId: t };
  } else if (READ_OPS.has(operation) || operation.startsWith('find') ||
             operation === 'update' || operation === 'delete') {
    next.where = { ...(next.where ?? {}), tenantId: t };
  }
  return next;
}

/** 构造 Prisma client 扩展；getCtx 从 CLS 读当前租户 */
export function createTenantExtension(getCtx: () => TenantCtx) {
  return Prisma.defineExtension({
    name: 'tenant-isolation',
    query: {
      $allModels: {
        async $allOperations({ model, operation, args, query }) {
          const ctx = getCtx();
          return query(buildTenantArgs(model as string, operation, args, ctx));
        },
      },
    },
  });
}
```

```typescript
import { INestApplication, Injectable, OnModuleInit, Logger } from '@nestjs/common';
import { PrismaClient } from '@prisma/client';
import { ClsService } from 'nestjs-cls';
import {
  CLS_PLATFORM_MODE_KEY, CLS_TENANT_KEY, PLATFORM_TENANT_ID,
} from '../tenancy/tenant.constants';
import { createTenantExtension } from './tenant-prisma.extension';

@Injectable()
export class PrismaService extends PrismaClient implements OnModuleInit {
  private readonly logger = new Logger(PrismaService.name);
  /** 带租户扩展的 client —— 业务代码应使用 this.tenant.* */
  public readonly tenant: ReturnType<PrismaService['buildTenantClient']>;

  constructor(private readonly cls: ClsService) {
    super();
    this.tenant = this.buildTenantClient();
  }

  private buildTenantClient() {
    return this.$extends(
      createTenantExtension(() => ({
        tenantId: this.cls.get(CLS_TENANT_KEY) ?? PLATFORM_TENANT_ID,
        isPlatform: this.cls.get(CLS_PLATFORM_MODE_KEY) === true,
      })),
    );
  }

  async onModuleInit() {
    // ...保持原有重试逻辑不变...
  }
  async enableShutdownHooks(app: INestApplication) {
    process.on('beforeExit', async () => { await app.close(); });
  }
}
```

> **迁移策略（关键）**：业务代码现在调用 `prisma.user.findMany(...)`。要让隔离生效，需逐步把业务调用从 `this.prisma.<model>` 改为 `this.prisma.tenant.<model>`。为降低风险，采用「**默认即隔离**」：把 `PrismaService` 默认导出的实例本身就是扩展后的——但 `$extends` 不可原地改写 this。折中方案见 Step 6。

```typescript
  // 在 constructor 末尾返回 Proxy，使 this.prisma.user 等同 this.tenant.user
  // 同时保留 $extends/$transaction/$queryRaw 等顶层方法走原 client
  // 实现：
  return new Proxy(this, {
    get(target, prop, receiver) {
      if (typeof prop === 'string' && /^[a-z]/.test(prop) &&
          prop in (target as any).tenant) {
        return (target as any).tenant[prop];
      }
      return Reflect.get(target, prop, receiver);
    },
  });
```

```typescript
// backend/src/tenancy/host.util.ts
import { domainToASCII } from 'node:url';
export function normalizeHost(raw: string | undefined | null): string | null {
  if (!raw) return null;
  let h = String(raw).trim().toLowerCase();
  if (!h) return null;
  h = h.split(',')[0].trim();      // x-forwarded-host 可能逗号分隔，取第一个
  h = h.replace(/:\d+$/, '');      // 去端口
  h = h.replace(/\.+$/, '');       // 去尾点
  if (!h) return null;
  const ascii = domainToASCII(h);  // punycode
  return ascii || h;
}
```

```typescript
// backend/src/tenancy/tenant-resolver.middleware.ts
import { Injectable, NestMiddleware, NotFoundException } from '@nestjs/common';
import { ClsService } from 'nestjs-cls';
import { ConfigService } from '@nestjs/config';
import { PrismaService } from '../prisma/prisma.service';
import { CLS_TENANT_KEY, PLATFORM_TENANT_ID } from './tenant.constants';
import { normalizeHost } from './host.util';

@Injectable()
export class TenantResolverMiddleware implements NestMiddleware {
  // host -> tenantId 缓存（生产可换 Redis；进程内 Map 起步）
  private cache = new Map<string, string>();
  constructor(
    private readonly cls: ClsService,
    private readonly prisma: PrismaService,
    private readonly config: ConfigService,
  ) {}

  async use(req: any, _res: any, next: () => void) {
    const trustForwarded = this.config.get('TRUST_FORWARDED_HOST') === 'true';
    const rawHost = (trustForwarded && req.headers['x-forwarded-host'])
      ? req.headers['x-forwarded-host']
      : req.headers['host'];
    const host = normalizeHost(rawHost);

    let tenantId = host ? this.cache.get(host) : undefined;
    if (host && !tenantId) {
      const row = await (this.prisma as any).tenantDomain.findUnique({ where: { host } });
      if (row) { tenantId = row.tenantId; this.cache.set(host, tenantId); }
    }

    if (!tenantId) {
      const devFallback = this.config.get('TENANT_UNKNOWN_HOST_FALLBACK') === 'platform';
      if (devFallback) tenantId = PLATFORM_TENANT_ID;       // 仅开发
      else throw new NotFoundException('站点未配置');         // 生产拒绝（codex#7）
    }
    this.cls.set(CLS_TENANT_KEY, tenantId);
    next();
  }
}
```

> `tenantDomain.findUnique` 走原 client（白名单表，扩展不注入）。缓存失效：子项目 3 增删域名时清缓存；基座用进程内 Map + 重启失效即可。

```javascript
// backend/scripts/check-raw-sql-tenant.mjs
// 扫描 src 下所有 $queryRaw/$executeRaw，未在同段含 tenant_id / ALLOW_RAW_NO_TENANT 注释则报错
import { readdirSync, readFileSync, statSync } from 'node:fs';
import { join } from 'node:path';
const ROOT = new URL('../src', import.meta.url).pathname;
const offenders = [];
function walk(dir) {
  for (const f of readdirSync(dir)) {
    const p = join(dir, f);
    if (statSync(p).isDirectory()) walk(p);
    else if (p.endsWith('.ts')) {
      const src = readFileSync(p, 'utf8');
      const re = /\$(queryRaw|executeRaw)/g; let m;
      while ((m = re.exec(src))) {
        const seg = src.slice(m.index, m.index + 400);
        if (!/tenant_id|ALLOW_RAW_NO_TENANT/i.test(seg)) {
          offenders.push(`${p}: ${m[0]}`);
        }
      }
    }
  }
}
walk(ROOT);
if (offenders.length) {
  console.error('Raw SQL without tenant_id / allow comment:\n' + offenders.join('\n'));
  process.exit(1);
}
```
